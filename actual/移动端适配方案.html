# 基于 postcss-pxtorem

# 安装依赖

// postcss-pxtorem 用于将单位转化为 rem
npm install postcss-pxtorem -D
// amfe-flexible 用于设置 rem 基准值
npm i -S amfe-flexible

# 使用

# 配置 postcss-pxtorem

在根目录新建一个 postcss.config.js 文件，配置如下

module.exports = {
  plugins: {
    autoprefixer: {
      overrideBrowserslist: ["Android >= 4.0", "iOS >= 8"]
      // 在vant官网这里是 browsers,但是运行会报postcss配置警告，所以改成 overrideBrowserslist,如果换成这个反而报错，换回browsers
    },
    "postcss-pxtorem": {
      rootValue: 37.5, //结果为：设计稿元素尺寸/37.5，比如元素宽375px,最终页面会换算成 10rem
      propList: ["*"]
    }
  }
};

# 在 main.js 引入 lib-flexible

import "amfe-flexible/index.js";

# index.html 设置 meta

<meta
  name="viewport"
  content="width=device-width,initial-scale=1,user-scalable=0,minimum-scale=1.0,maximum=scale=1.0"
/>

# 手撸 js

# 设置 rem 函数

function setRem() {
  // 320 默认大小16px; 320px = 20rem ;每个元素px基础上/16
  let htmlWidth =
    document.documentElement.clientWidth || document.body.clientWidth; //得到html的Dom元素
  let htmlDom = document.getElementsByTagName("html")[0]; //设置根元素字体大小
  htmlDom.style.fontSize = htmlWidth / 20 + "px";
} // 初始化setRem();// 改变窗口大小时重新设置 rem

window.onresize = function() {
  setRem();
};

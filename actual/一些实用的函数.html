# 获取 url 参数

getUrlParam(name) { // 获取url参数
    var reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)", "i");
    var r = window.location.search.substr(1).match(reg);
    if (r != null) return unescape(r[2]);
    return null;
},

# 获得 & 后面的值

getStringParam(name, str) { // 获得 & 后面的值 , 参数 1 为需要查询的 key，参数 2 为用来匹配 key 的 url
  var reg = new RegExp("(^|&)" + name + "=([^&]_)(&|\$)", "i");
  var r = str.substr(1).match(reg);
  if (r != null) return unescape(r[2]);
  return null;
},

# 获取当前页面 url

getUrl(path) { // 获取当前页面 url
  let ip = window.location.host; //获取服务器 ip 地址
  let protocolStr = document.location.protocol; // 协议头
  // process.env.NODE_ENV 获取你的环境变量 path就写你要跳的路由
  let url = `${protocolStr}//${ip}/${process.env.NODE_ENV}/${path}`;
  return url
},

# 获取年月日时分秒

formatDateNew(time) {
  if (!time) return ''
  if (typeof time == 'string') { // 兼容 ios
    time = time && time.indexOf('-') != -1 ? time.replace(/-/g, "/") : time;
  }
  let d = new Date(time);
  let year = d.getFullYear();
  let mouth = this.formatNumber(d.getMonth() + 1);
  let date = this.formatNumber(d.getDate());
  let hours = this.formatNumber(d.getHours());
  let minute = this.formatNumber(d.getMinutes());
  let seconds = this.formatNumber(d.getSeconds());
  return {
    year,
    mouth,
    date,
    hours,
    minute,
    seconds
  }
},

# 判断页面的打开环境

isEquipment() { //判断页面的打开环境
  let UA = navigator.userAgent,
  isAndroid = /android|adr|linux/gi.test(UA),
  isIOS = /iphone|ipod|ipad/gi.test(UA) && !isAndroid,
  isBlackBerry = /BlackBerry/i.test(UA),
  isWindowPhone = /IEMobile/i.test(UA),
  isMobile = isAndroid || isIOS || isBlackBerry || isWindowPhone,
  isApp = this.hasNative(); // 这里需要配合你们自己的 app 方法，自己定义一下！
  return {
    isAndroid: isAndroid, //是安卓
    isIOS: isIOS, //是苹果
    isMobile: isMobile, //是手机
    isWeixin: /MicroMessenger/gi.test(UA), //是微信
    isQQ: /QQ/gi.test(UA), //是 QQ
    isPC: !isMobile, //是 PC
    isWeibo: /WeiBo/gi.test(UA),
    isApp: isApp || false //是 app
  }
},

# 对 url 进行编码

utf8_decode (utftext) { //对 url 进行编码
  var string = '';
  var i = c = c1 = c2 = 0;
  //获取 URL 所有字符
  while (i < utftext.length) {
  //获取 URL 并将 URL 中所有 Unicode 编码获取
    c = utftext.charCodeAt(i);
    //对 Unicode 编码进行处理转化成字符串
    if (c < 128) {
    string += String.fromCharCode(c);
    i++;
    }else if ((c < 191) && (c < 224)) {
      c2 = utftext.charCodeAt(i + 1);
      string += String.fromCharCode(((c & 31) << 6) | (c2 & 63));
      i += 2;
    }else {
      c2 = utftext.charCodeAt(i + 1);
      c3 = utftext.charCodeAt(i + 2);
      string += String.fromCharCode(((c & 15) << 12) | ((c2 & 63) << 6) | (c3 & 63));
      i += 3;
    }
  }
  return string;
}

# 去除指定参数

funcUrlDel(names,url) { // 参数一为取出的参数，可以是string，也可以是数组; 参数二为指定的url
      if (typeof (names) == 'string') {
        names = [names];
      }
      var loca = url;
      var obj = {}
      var arr = loca.search.substr(1).split("&");
      //获取参数转换为object
      for (var i = 0; i < arr.length; i++) {
        arr[i] = arr[i].split("=");
        obj[arr[i][0]] = arr[i][1];
      };
      //删除指定参数
      for (var i = 0; i < names.length; i++) {
        delete obj[names[i]];
      }
      //重新拼接url
      var url = loca.origin + loca.pathname + "?" + JSON.stringify(obj).replace(/[\"\{\}]/g, "").replace(/\:/g, "=").replace(/\,/g, "&");
      console.log('截取后', url)
      return url;
}
